Installation

Node.js runs JavaScript outside the browser. Download the LTS version from the official site or use a version manager.

# Option 1: Download from nodejs.org (LTS recommended)

# Option 2: Using nvm (Node Version Manager)
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.7/install.sh | bash
nvm install --lts  # Install latest LTS
nvm use --lts    # Switch to LTS version

# Verify installation
node --version    # e.g., v20.11.0
npm --version    # e.g., 10.2.4

# Option 3: Package manager (Linux)
sudo apt install nodejs npm  # Debian/Ubuntu
sudo pacman -S nodejs npm   # Arch Linux

npm & package.json

npm is the default package manager for Node.js. The package.json file tracks dependencies, scripts, and project metadata.

# Initialize a new project
npm init -y     # Creates package.json with defaults

# Install packages
npm install express # Saves to dependencies
npm install -D nodemon # Saves to devDependencies
npm install -g pm2  # Global install (available system-wide)

# Uninstall packages
npm uninstall express

# package.json structure
{
 "name": "my-api",
 "version": "1.0.0",
 "scripts": {
  "start": "node server.js",
  "dev": "nodemon server.js"
 },
 "dependencies": {
  "express": "^4.18.2"
 },
 "devDependencies": {
  "nodemon": "^3.0.0"
 }
}
node_modules should never be committed to Git. Add node_modules/ to your .gitignore. The package-lock.json file should be committed to lock exact dependency versions.

File System Module

The fs module lets you read, write, delete, and watch files on the server's filesystem.

const fs = require("fs");
const path = require("path");

// Read file (async)
fs.readFile("data.txt", "utf8", (err, data) => {
 if (err) throw err;
 console.log(data);
});

// Read file (sync — use sparingly)
const content = fs.readFileSync("data.txt", "utf8");

// Write file
fs.writeFileSync("output.txt", "Hello, Node.js!");

// Append to file
fs.appendFileSync("log.txt", `Request at ${new Date()}\n`);

// Check if file exists
if (fs.existsSync("data.txt")) {
 console.log("File exists");
}

// Create directory
fs.mkdirSync("uploads", { recursive: true });

// Read directory contents
const files = fs.readdirSync("./");
console.log(files);

// Using promises (modern approach)
const fsPromises = require("fs").promises;
async function readData() {
 const data = await fsPromises.readFile("data.txt", "utf8");
 return data;
}

Express.js Basics

Express is the most popular web framework for Node.js. It provides routing, middleware, and request/response handling.

const express = require("express");
const app = express();
const port = process.env.PORT || 3000;

// Middleware — runs on every request
app.use(express.json());      // Parse JSON bodies
app.use(express.urlencoded({ extended: true })); // Parse form data

// Routes
app.get("/", (req, res) => {
 res.send("Hello, World!");
});

app.get("/api/users", (req, res) => {
 res.json([{ id: 1, name: "Alice" }, { id: 2, name: "Bob" }]);
});

app.post("/api/users", (req, res) => {
 const user = req.body;  // Access parsed JSON body
 res.status(201).json({ message: "User created", user });
});

// Route parameters
app.get("/api/users/:id", (req, res) => {
 const id = req.params.id;
 res.json({ id, name: `User ${id}` });
});

// Start server
app.listen(port, () => {
 console.log(`Server running on http://localhost:${port}`);
});

REST APIs

RESTful APIs follow a standard pattern of endpoints for CRUD operations. Here's a complete example with in-memory data.

const express = require("express");
const app = express();
app.use(express.json());

let items = [
 { id: 1, name: "Learn Node.js", done: false },
 { id: 2, name: "Build an API", done: false }
];
let nextId = 3;

// GET /api/items — list all
app.get("/api/items", (req, res) => res.json(items));

// GET /api/items/:id — get one
app.get("/api/items/:id", (req, res) => {
 const item = items.find(i => i.id === parseInt(req.params.id));
 if (!item) return res.status(404).json({ error: "Not found" });
 res.json(item);
});

// POST /api/items — create
app.post("/api/items", (req, res) => {
 const item = { id: nextId++, name: req.body.name, done: false };
 items.push(item);
 res.status(201).json(item);
});

// PUT /api/items/:id — update
app.put("/api/items/:id", (req, res) => {
 const item = items.find(i => i.id === parseInt(req.params.id));
 if (!item) return res.status(404).json({ error: "Not found" });
 Object.assign(item, req.body);
 res.json(item);
});

// DELETE /api/items/:id — delete
app.delete("/api/items/:id", (req, res) => {
 items = items.filter(i => i.id !== parseInt(req.params.id));
 res.status(204).send();
});

app.listen(3000);

Environment Variables

Environment variables keep secrets and configuration out of your code. Use the dotenv package to load a .env file.

# Install dotenv
npm install dotenv

# .env file (never commit this!)
PORT=3000
DB_URL=mongodb://localhost:27017/myapp
JWT_SECRET=your-secret-key-here
API_KEY=abc123

# server.js
require("dotenv").config();

const port = process.env.PORT || 3000;
const dbUrl = process.env.DB_URL;
const apiKey = process.env.API_KEY;

app.listen(port, () => {
 console.log(`Server running on port ${port}`);
 console.log(`Database: ${dbUrl}`);
});
Security: Never commit .env files to Git! Add .env to your .gitignore. Provide a .env.example file in your repo with placeholder values so other developers know what variables are needed.

Deployment

Deploying a Node.js app is straightforward. Here are the most popular platforms:

PlatformHow It Works
RenderConnect GitHub repo, set build command, set start command — free tier available
RailwayAuto-detects Node.js, deploys from GitHub with zero config
Fly.ioUses Docker with a generated config, generous free tier
VPS (DigitalOcean, Linode)Full control — use pm2 or systemd to keep the app running
 
# Using pm2 for process management (VPS)
npm install -g pm2
pm2 start server.js --name "my-app"
pm2 save
pm2 startup  # Auto-start on server reboot

# Check status
pm2 list
pm2 logs
pm2 restart my-app
