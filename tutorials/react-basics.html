Components & JSX

React apps are built from reusable components. JSX is a syntax extension that looks like HTML but compiles to JavaScript function calls.

// Function component (modern React)
function Welcome({ name }) {
 return <h1>Hello, {name}!</h1>;
}

// Arrow function component
const Card = ({ title, children }) => {
 return (
  <div className="card">
   <h2>{title}</h2>
   <div>{children}</div>
  </div>
 );
};

// Using components
function App() {
 return (
  <div>
   <Welcome name="Alice" />
   <Card title="My Card">
    <p>This is the card body.</p>
   </Card>
  </div>
 );
}

// JSX rules: className instead of class, camelCase properties
// Expressions in {curly braces}, self-closing tags must end with /

Props vs State

Props are read-only data passed from parent to child. State is data that changes over time within a component.

// Props — passed like HTML attributes
function UserProfile({ username, avatar, isOnline }) {
 return (
  <div className="profile">
   <img src={avatar} alt={username} />
   <span>{username}</span>
   <span className={isOnline ? "online" : "offline"}>
    {isOnline ? "Online" : "Offline"}
   </span>
  </div>
 );
}

// Default props
function Button({ children, variant = "primary" }) {
 return <button className={`btn btn-${variant}`}>{children}</button>;
}

useState & useEffect

Hooks let function components use state and lifecycle features. useState manages local state; useEffect handles side effects.

import { useState, useEffect } from "react";

function Counter() {
 // useState returns [value, setter]
 const [count, setCount] = useState(0);
 const [name, setName] = useState("");

 // useEffect — runs after render
 useEffect(() => {
  document.title = `Count: ${count}`;
 }, [count]); // Only re-run when `count` changes

 // useEffect with empty deps — runs once (mount)
 useEffect(() => {
  console.log("Component mounted");
  return () => console.log("Component unmounted"); // Cleanup
 }, []);

 return (
  <div>
   <p>Count: {count}</p>
   <button onClick={() => setCount(c => c + 1)}>+</button>
   <button onClick={() => setCount(c => c - 1)}>-</button>
   <input value={name} onChange={e => setName(e.target.value)} />
  </div>
 );
}
State updates are async. If you need the previous state to compute the next value, use the callback form: setCount(prev => prev + 1). Never mutate state directly — always use the setter function.

Event Handling

React events are named with camelCase and passed as props. Event handlers receive a synthetic event that works consistently across browsers.

function Form() {
 const [input, setInput] = useState("");

 const handleSubmit = (e) => {
  e.preventDefault(); // Prevent page reload
  console.log("Submitted:", input);
 };

 const handleChange = (e) => {
  setInput(e.target.value);
 };

 const handleKeyDown = (e) => {
  if (e.key === "Enter" && e.ctrlKey) {
   console.log("Ctrl+Enter pressed");
  }
 };

 return (
  <form onSubmit={handleSubmit}>
   <input
    type="text"
    value={input}
    onChange={handleChange}
    onKeyDown={handleKeyDown}
   />
   <button type="submit">Submit</button>
  </form>
 );
}

// Passing arguments to handlers
<button onClick={() => handleDelete(item.id)}>Delete</button>

React Router

React Router adds client-side navigation to your app without full page reloads. Install it with your package manager.

// Install: npm install react-router-dom

import { BrowserRouter, Routes, Route, Link, useParams } from "react-router-dom";

function Home() { return <h1>Home</h1>; }
function About() { return <h1>About</h1>; }

function User() {
 const { id } = useParams(); // Access route parameters
 return <h1>User {id}</h1>;
}

function NotFound() { return <h1>404 — Not Found</h1>; }

function App() {
 return (
  <BrowserRouter>
   <nav>
    <Link to="/">Home</Link>
    <Link to="/about">About</Link>
    <Link to="/users/1">User 1</Link>
   </nav>

   <Routes>
    <Route path="/" element={<Home />} />
    <Route path="/about" element={<About />} />
    <Route path="/users/:id" element={<User />} />
    <Route path="*" element={<NotFound />} />
   </Routes>
  </BrowserRouter>
 );
}

Project with Vite

Vite is the recommended way to start a React project. It's fast, modern, and requires zero configuration.

# Create a new React project with Vite
npm create vite@latest my-app -- --template react
cd my-app
npm install
npm run dev     # Start dev server (localhost:5173)

# Project structure
my-app/
 index.html     # Entry HTML file
 src/
  main.jsx     # React entry point
  App.jsx     # Root component
  App.css     # Root styles
  index.css    # Global styles
 vite.config.js   # Vite configuration
 package.json

# Build for production
npm run build     # Output in dist/ folder

# Preview production build
npm run preview
// src/main.jsx — entry point
import React from "react";
import ReactDOM from "react-dom/client";
import App from "./App";
import "./index.css";

ReactDOM.createRoot(document.getElementById("root")).render(
 <React.StrictMode>
  <App />
 </React.StrictMode>
);
