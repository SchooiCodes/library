Advanced SQL Queries & Optimization

Master window functions, CTEs, indexing, and query performance tuning.

Table of Contents (5 sections)

  • Window Functions

Window Functions

SELECT name, department, salary,
    ROW_NUMBER() OVER (PARTITION BY department ORDER BY salary DESC) as rank
FROM employees;

-- Running total
SELECT date, amount,
    SUM(amount) OVER (ORDER BY date) as running_total
FROM transactions;

Common Table Expressions

WITH sales_summary AS (
    SELECT category, SUM(amount) as total_sales
    FROM orders GROUP BY category
)
SELECT * FROM sales_summary ORDER BY total_sales DESC;

Indexing Strategies

-- B-tree index
CREATE INDEX idx_email ON users(email);

-- Composite index (order matters!)
CREATE INDEX idx_state_city ON addresses(state, city);

-- Partial index
CREATE INDEX idx_active ON users(email) WHERE active = true;

Query Optimization

-- Use EXISTS instead of IN
SELECT * FROM customers c
WHERE EXISTS (SELECT 1 FROM orders o WHERE o.customer_id = c.id);

-- Avoid functions in WHERE
WHERE created_at >= '2024-01-01' AND created_at < '2024-01-02'

Advanced Joins

-- Lateral join
SELECT c.name, latest.total as last_order
FROM customers c
LEFT JOIN LATERAL (
    SELECT total FROM orders
    WHERE customer_id = c.id
    ORDER BY created_at DESC LIMIT 1
) latest ON true;
