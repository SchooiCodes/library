What is Load Balancing?

Load balancing distributes incoming network traffic across multiple backend servers to ensure no single server bears too much demand. This improves availability, reliability, and performance. A reverse proxy similarly sits in front of backend servers but often adds SSL termination, caching, and routing capabilities.

Common use cases: Distribute traffic between multiple app servers, Terminate SSL/TLS certificates centrally, Route requests to different services by domain/path, Cache static content, Rate limiting and DDoS protection, Canary releases and blue-green deployments.

Nginx as Reverse Proxy

# /etc/nginx/sites-available/app
server {
  listen 80;
  server_name app.example.com;
  return 301 https://$server_name$request_uri;
}

server {
  listen 443 ssl http2;
  server_name app.example.com;

  ssl_certificate /etc/letsencrypt/live/app.example.com/fullchain.pem;
  ssl_certificate_key /etc/letsencrypt/live/app.example.com/privkey.pem;

  location / {
    proxy_pass http://127.0.0.1:3000;
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;

    # WebSocket support
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection "upgrade";
  }

  location /api/ {
    proxy_pass http://127.0.0.1:8080;
    proxy_set_header Host $host;
  }

  location /static/ {
    alias /var/www/app/static;
    expires 30d;
    add_header Cache-Control "public, immutable";
  }
}

HAProxy Load Balancer

# /etc/haproxy/haproxy.cfg
global
  log /dev/log local0
  maxconn 4096
  user haproxy
  group haproxy

defaults
  log global
  mode http
  option httplog
  option dontlognull
  retries 3
  timeout connect 5000ms
  timeout client 50000ms
  timeout server 50000ms

frontend web_front
  bind *:80
  bind *:443 ssl crt /etc/ssl/certs/haproxy.pem
  redirect scheme https if !{ ssl_fc }

  # Routing by domain
  use_backend app_servers if { hdr(Host) -i app.example.com }
  use_backend api_servers if { hdr(Host) -i api.example.com }
  default_backend app_servers

backend app_servers
  balance roundrobin
  option forwardfor
  http-request set-header X-Forwarded-Proto https if { ssl_fc }

  server app1 192.168.1.10:3000 check
  server app2 192.168.1.11:3000 check
  server app3 192.168.1.12:3000 check

backend api_servers
  balance leastconn # Better for long-lived connections
  server api1 192.168.1.10:8080 check
  server api2 192.168.1.11:8080 check

Caddy (Auto-HTTPS)

Caddy is the easiest reverse proxy — it automatically provisions and renews SSL certificates via Let's Encrypt. No certbot, no manual SSL config.

# Caddyfile
app.example.com {
  reverse_proxy localhost:3000
}

api.example.com {
  reverse_proxy /api/* localhost:8080 {
    header_up X-Real-IP {remote_host}
  }
}

static.example.com {
  root * /var/www/static
  file_server
}

# Multiple backends with load balancing
app.example.com {
  reverse_proxy {
    to localhost:3000 localhost:3001 localhost:3002
    lb_policy round_robin
    health_uri /health
    health_interval 10s
  }
}

# Path-based routing
example.com {
  route /api/* {
    reverse_proxy localhost:8080
  }
  route /* {
    reverse_proxy localhost:3000
  }
}

SSL/TLS Termination

Offloading SSL processing to the reverse proxy reduces backend server load and centralizes certificate management.

# Let's Encrypt with Nginx
sudo apt install certbot python3-certbot-nginx
sudo certbot --nginx -d app.example.com -d api.example.com

# Auto-renewal (already configured by certbot)
sudo certbot renew --dry-run

# HAProxy with combined PEM
sudo cat /etc/letsencrypt/live/app.example.com/fullchain.pem \
 /etc/letsencrypt/live/app.example.com/privkey.pem \
 > /etc/ssl/certs/haproxy.pem

# Add renewal hook to combine certs
# /etc/letsencrypt/renewal-hooks/deploy/haproxy.sh
#!/bin/bash
cat /etc/letsencrypt/live/app.example.com/fullchain.pem \
 /etc/letsencrypt/live/app.example.com/privkey.pem \
 > /etc/ssl/certs/haproxy.pem
systemctl reload haproxy

Health Checks & Monitoring

Health checks ensure traffic is only routed to healthy backend servers. Both active (periodic probes) and passive (monitoring real traffic) checks are important.

# Nginx passive health check (proxies next upstream on failure)
upstream backend {
  server 192.168.1.10:3000 max_fails=3 fail_timeout=30s;
  server 192.168.1.11:3000 max_fails=3 fail_timeout=30s;
}

# HAProxy active health check
backend app_servers
  option httpchk GET /health HTTP/1.1\r\nHost:\ app.example.com
  server app1 192.168.1.10:3000 check inter 5s fall 3 rise 2
  server app2 192.168.1.11:3000 check inter 5s fall 3 rise 2

# Health endpoint on your app (Node.js example)
app.get('/health', async (req, res) => {
 const dbOk = await checkDatabase();
 const status = dbOk ? 200 : 503;
 res.status(status).json({ status: status === 200 ? 'ok' : 'degraded' });
});

Load Balancing Strategies

AlgorithmDescriptionBest For
Round RobinDistributes requests evenly in rotationEqual-capacity servers, simple setups
Least ConnectionsSends to server with fewest active connectionsLong-lived connections, variable request times
IP HashMaps client IP to a specific serverSession persistence without cookies
WeightedAssigns ratios to servers (e.g., 3:1 for powerful vs weak)Heterogeneous server capacities
RandomPicks a random serverTesting, avoiding patterns
